void mainImage( out vec4 fragColor, in vec2 fragCoord )
{
    //get inverse resolution
    sCoord resInv 	= 1.0 / iResolution.xy;
    
    //get normalized coords
    sCoord uv 		= fragCoord * resInv;
    
    //gather reset input
    float reset = texture(iChannel1, uv).w;
    
    //get previous coords
    sCoord current 	= texture(iChannel0, uv).xy * reset;
    sCoord origin 	= texture(iChannel0, uv).zw * reset;
    
    //track  input if mouse down
    if(iMouse.z >= 0.0)
    {
        //get current position of mouse
    	sCoord click 	= iMouse.zw * resInv;
        sCoord mouse 	= iMouse.xy * resInv; 
        
        //get position relative to the center
        sCoord center 	= click + (click - sCoord(0.5, 0.5));
        sCoord mouseRev = mouse + (mouse - sCoord(0.5, 0.5));
        
        //get final position relative to the origin
        sCoord RadPos	= origin - (mouseRev - center);
        
        //return new position
        fragColor 		= vec4(RadPos, origin);
    }
    
    else
    {
        //only display old coords if not using
    	fragColor 		= vec4(current, current);
    }
}
