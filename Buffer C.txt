void mainImage( out vec4 fragColor, in vec2 fragCoord )
{   
    //get inverse resolution
    vec2 invRes = 1.0 / iChannelResolution[0].xy;
    
    //guassian blur
    fragColor = guassianBlur(iChannel0, fragCoord, invRes);
    
    //edge detection
    fragColor = edgeDetection(iChannel0, fragCoord, invRes);
}
